Build Setup

# 初始化项目
#使用之前确保已安装node 
# npm i  gitbook -g 或 cnpm i gitbook -g // 全局安装gitbook 

git clone https://github.com/emjio/gitbook-start.git

cd gitbook-start

# 安装依赖
gitbook instart 

# 开发时构建
gitbook serve 
浏览器打开 http://localhost:4000/ 预览

# 打包构建
gitbook build
